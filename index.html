Mini Event Framework

It allows javascript objects to bind and listen to events.

Class - EventEmitter

Methods

  • bind
  • unbind
  • trigger

Sample Usage

function Model(){};

eventify(Model,EventEmitter);

var myModel=new Model();

function foo(){
	console.log(" OMG ");
}

myModel.bind("hello",function(){
	console.log("1st function called");
})

myModel.bind("hello",function(){
	console.log("2nd function called");
})

myModel.bind("hello",foo);

myModel.trigger("hello",{"name":"devang"});

myModel.unbind("hello",foo);

myModel.trigger("hello",{"name":"devang"});
